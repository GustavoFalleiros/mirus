59  void RTC::read_rtc()
60  {
61  unsigned char last_second;
62  unsigned char last_minute;
63  unsigned char last_hour;
64  unsigned char last_day;
65  unsigned char last_month;
66  unsigned char last_year;
67  unsigned char last_century;
68  unsigned char registerB;
69 
70  // Note: This uses the "read registers until you get the same values twice in a row" technique
71  // to avoid getting dodgy/inconsistent values due to RTC updates
72 
73  // Make sure an update isn't in progress
74  while (RTC::get_update_in_progress_flag());
75 
76  second = RTC::get_rtc_register(0x00);
77  minute = RTC::get_rtc_register(0x02);
78  hour = RTC::get_rtc_register(0x04);
79  day = RTC::get_rtc_register(0x07);
80  month = RTC::get_rtc_register(0x08);
81  year = RTC::get_rtc_register(0x09);
82 
83  if(century_register != 0)
84  century = RTC::get_rtc_register(century_register);
85 
86  do
87  {
88  last_second = second;
89  last_minute = minute;
90  last_hour = hour;
91  last_day = day;
92  last_month = month;
93  last_year = year;
94  last_century = century;
95 
96  // Make sure an update isn't in progress
97  while (RTC::get_update_in_progress_flag());
98 
99  second = RTC::get_rtc_register(0x00);
100  minute = RTC::get_rtc_register(0x02);
101  hour = RTC::get_rtc_register(0x04);
102  day = RTC::get_rtc_register(0x07);
103  month = RTC::get_rtc_register(0x08);
104  year = RTC::get_rtc_register(0x09);
105 
106  if(century_register != 0)
107  century = RTC::get_rtc_register(century_register);
108 
109  }
110  while ((last_second != second)
111  || (last_minute != minute)
112  || (last_hour != hour)
113  || (last_day != day)
114  || (last_month != month)
115  || (last_year != year)
116  || (last_century != century));
117 
118  registerB = RTC::get_rtc_register(0x0B);
119 
120  // Convert BCD to binary values if necessary
121  if (!(registerB & 0x04))
122  {
123  second = (second & 0x0F) + ((second / 16) * 10);
124  minute = (minute & 0x0F) + ((minute / 16) * 10);
125  hour = ((hour & 0x0F) + (((hour & 0x70) / 16) * 10)) | (hour & 0x80);
126  day = (day & 0x0F) + ((day / 16) * 10);
127  month = (month & 0x0F) + ((month / 16) * 10);
128  year = (year & 0x0F) + ((year / 16) * 10);
129 
130  if(century_register != 0)
131  century = (century & 0x0F) + ((century / 16) * 10);
132  }
133 
134  // Convert 12 hour clock to 24 hour clock if necessary
135  if (!(registerB & 0x02) && (hour & 0x80))
136  {
137  hour = ((hour & 0x7F) + 12) % 24;
138  }
139 
140  // Calculate the full (4-digit) year
141  if(century_register != 0)
142  year += century * 100;
143  else
144  {
145  year += (CURRENT_YEAR / 100) * 100;
146 
147  if(year < CURRENT_YEAR)
148  year += 100;
149  }
150  }
